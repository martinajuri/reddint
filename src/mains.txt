Fecha* ingresarFecha()
{
    int d, m, a;
    cout <<"Dia: ";
    cin >> d;
    cout <<"Mes: ";
    cin >> m;
    cout << "Año: ";
    cin >> a;
    return new Fecha(d,m,a);
}

Usuario* ingresarUsuario()
{
    string nombre;
    cout << "Nombre: ";
    getline(cin, nombre);
    return new Usuario(nombre);
}

Contenido* ingresarPublicacion(Fecha* fecha, Usuario* usuario)
{
    string titulo, cuerpo;

    getline(cin.ignore(), titulo);
    cout << "Cuerpo: ";
    getline(cin, cuerpo);

}
